# S1 Core Library — Web (Nuxt) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** The web library experience — a `/library` list that filters by read-state/favorite/tag and full-text `q`, saves a link from an inline bar (appears instantly as `pending`, fills in live via SSE), opens an item’s metadata detail, toggles read-state/favorite, adds/removes tags, deletes — plus a minimal settings page. All HTTP through a typed `useApi`, all state in `useState`-backed composables, all strings via per-component `<i18n>` blocks.

**Architecture:** Composable state (no Pinia in this repo): `useItems`/`useTags` hold `useState` singletons and call `useApi` (a thin typed `$fetch` wrapper over the REST API, typed from the generated `@app/specs`). The list-mutation and SSE-routing logic lives in **pure, unit-tested utils** (`utils/items-store.ts`); composables wrap them. `useEvents` opens a native `EventSource` to `/api/v1/events` and re-fetches changed items, driving the `pending → ready` transition without a refresh. Pages compose the S1-ui composites (`AppItemCard`, `AppFilterBar`, `AppEmptyState`, `AppSkeleton`, `AppTagChip`).

**Tech Stack:** Nuxt 4 (SPA, `ssr:false`), `@nuxt/ui` v4, `@nuxtjs/i18n` (en/ru, per-component `<i18n>` blocks), `@app/ui`, Vitest (`happy-dom`, `globals:true`).

**Depends on:** **S1-ui** (the six composites) and **S1-auth** (the `auth` middleware + that `/welcome` exists; `/library` is the route its guard points at). S1-backend is shipped. **Out of scope:** the item-detail reader / full-article fetch machine (S2); a standalone add-link route (the save flow is an inline bar + modal on the library page).

## Conventions & contracts

- **API base:** `useRuntimeConfig().public.apiBaseUrl` already ends in `/api/v1`; `useApi` paths are spec paths minus that prefix (`/items`, `/tags`, `/events`). Auth is the session cookie via `credentials:'include'`.
- **Generated types:** `import type { components } from '@app/specs'` → `components['schemas']['Item' | 'ItemList' | 'Tag' | 'TagList' | 'SaveItemRequest' | 'UpdateItemRequest' | 'ReadState' | 'ItemStatus']`. Never hand-declare wire types.
- **SSE envelope (confirm against the S0 writer):** one `message` event whose `data` is JSON `{ type, data }`. The shipped backend emits **`{ id }`** payloads (not the full item) for `item.created`/`item.updated`/`item.deleted` — so the client re-fetches the changed item by id. `job.updated` is ignored by the list.
- **i18n:** `const { t } = useI18n({ useScope: 'local' })` + an `<i18n lang="json">` block with `en` and `ru` only.
- **Tokens / BEM:** pages use `.page-<name>` scoped SCSS on `var(--token)` names (same token set as S1-ui). No Tailwind utilities in markup, no inline literal styles, no `!important`.
- **Protected pages:** `definePageMeta({ middleware: 'auth' })` (middleware ships in S1-auth).

**Commands:** `pnpm --filter @app/web test` · `pnpm --filter @app/web typecheck` · `pnpm --filter @app/web lint --fix` · `pnpm stylelint:fix` · `pnpm format`.

`apps/web/app/utils/items-store.ts`:

```ts
import type { components } from '@app/specs';

type Item = components['schemas']['Item'];

export type EventAction =
  | { kind: 'refetch'; id: string }
  | { kind: 'remove'; id: string }
  | { kind: 'ignore' };

export function upsertItem(items: Item[], next: Item): Item[] {
  const idx = items.findIndex((i) => i.id === next.id);
  if (idx === -1) return [next, ...items];
  const copy = items.slice();
  copy[idx] = next;
  return copy;
}

export function removeItemById(items: Item[], id: string): Item[] {
  return items.filter((i) => i.id !== id);
}

export function routeEvent(evt: { type: string; data: { id?: string } }): EventAction {
  switch (evt.type) {
    case 'item.created':
    case 'item.updated':
      return evt.data.id ? { kind: 'refetch', id: evt.data.id } : { kind: 'ignore' };
    case 'item.deleted':
      return evt.data.id ? { kind: 'remove', id: evt.data.id } : { kind: 'ignore' };
    default:
      return { kind: 'ignore' };
  }
}
```

```ts
import type { AppFilterSegment } from '@app/ui';

import type { components } from '@app/specs';
import type { ItemsQuery } from './useApi';
import { removeItemById, upsertItem } from '~/utils/items-store';

type Item = components['schemas']['Item'];

interface Filters {
  segment: AppFilterSegment;
  q: string;
  tag: string | null;
}

function toQuery(f: Filters): ItemsQuery {
  const base: ItemsQuery = { q: f.q || undefined, tag: f.tag ?? undefined, limit: 20 };
  return f.segment === 'favorite' ? { ...base, favorite: true } : { ...base, readState: f.segment };
}

export function useItems() {
  const api = useApi();
  const items = useState<Item[]>('items:list', () => []);
  const nextCursor = useState<string | null>('items:cursor', () => null);
  const loading = useState<boolean>('items:loading', () => false);
  const filters = useState<Filters>('items:filters', () => ({
    segment: 'unread',
    q: '',
    tag: null,
  }));

  async function load(): Promise<void> {
    loading.value = true;
    try {
      const page = await api.getItems(toQuery(filters.value));
      items.value = page.items;
      nextCursor.value = page.nextCursor;
    } finally {
      loading.value = false;
    }
  }

  async function loadMore(): Promise<void> {
    if (!nextCursor.value) return;
    const page = await api.getItems({ ...toQuery(filters.value), cursor: nextCursor.value });
    items.value = [...items.value, ...page.items];
    nextCursor.value = page.nextCursor;
  }

  async function refetchOne(id: string): Promise<void> {
    try {
      items.value = upsertItem(items.value, await api.getItem(id));
    } catch {
      /* item may have been deleted between event and fetch — ignore */
    }
  }

  function removeLocal(id: string): void {
    items.value = removeItemById(items.value, id);
  }

  async function save(url: string): Promise<void> {
    items.value = upsertItem(items.value, await api.saveItem({ url }));
  }

  async function toggleFavorite(item: Item): Promise<void> {
    items.value = upsertItem(items.value, { ...item, favorite: !item.favorite }); // optimistic
    await api.updateItem(item.id, { favorite: !item.favorite });
  }

  async function setReadState(item: Item, readState: Item['readState']): Promise<void> {
    items.value = upsertItem(items.value, { ...item, readState }); // optimistic
    await api.updateItem(item.id, { readState });
  }

  async function remove(id: string): Promise<void> {
    removeLocal(id); // optimistic
    await api.deleteItem(id);
  }

  return {
    items,
    nextCursor,
    loading,
    filters,
    load,
    loadMore,
    refetchOne,
    removeLocal,
    save,
    toggleFavorite,
    setReadState,
    remove,
  };
}
```

### Task 4: `useEvents` — live SSE → store

**Files:** Create `apps/web/app/composables/useEvents.ts`.

- [ ] **Step 1: Implement** — open `EventSource`, route each envelope through `routeEvent`, refetch/remove via `useItems`. (The pure routing is already tested in Task 2; this is the side-effect shell.)

```ts
import { routeEvent } from '~/utils/items-store';

export function useEvents() {
  const store = useItems();
  const { public: pub } = useRuntimeConfig();
  let es: EventSource | null = null;

  function connect(): void {
    if (es) return;
    es = new EventSource(`${pub.apiBaseUrl}/events`, { withCredentials: true });
    es.onmessage = (e: MessageEvent<string>) => {
      let evt: { type: string; data: { id?: string } };
      try {
        evt = JSON.parse(e.data);
      } catch {
        return;
      }
      const action = routeEvent(evt);
      if (action.kind === 'refetch') void store.refetchOne(action.id);
      else if (action.kind === 'remove') store.removeLocal(action.id);
    };
  }

  function disconnect(): void {
    es?.close();
    es = null;
  }

  onMounted(connect);
  onBeforeUnmount(disconnect);
  return { connect, disconnect };
}
```

```vue
<script setup lang="ts">
  import { AppTagChip, AppButton, AppInput, AppStatusBadge } from '@app/ui';
  import { ref, onMounted } from 'vue';

  import type { components } from '@app/specs';

  definePageMeta({ middleware: 'auth' });
  const { t } = useI18n({ useScope: 'local' });
  const route = useRoute();
  const api = useApi();
  const id = route.params['id'] as string;

  type Item = components['schemas']['Item'];
  const item = ref<Item | null>(null);
  const newTag = ref('');

  async function refresh(): Promise<void> {
    item.value = await api.getItem(id);
  }
  onMounted(refresh);

  async function setReadState(readState: Item['readState']): Promise<void> {
    item.value = await api.updateItem(id, { readState });
  }
  async function toggleFavorite(): Promise<void> {
    if (item.value) item.value = await api.updateItem(id, { favorite: !item.value.favorite });
  }
  async function addTag(): Promise<void> {
    if (!newTag.value.trim()) return;
    item.value = await api.addItemTag(id, newTag.value.trim());
    newTag.value = '';
  }
  async function removeTag(slug: string): Promise<void> {
    await api.removeItemTag(id, slug);
    await refresh();
  }
  async function remove(): Promise<void> {
    await api.deleteItem(id);
    await navigateTo('/library');
  }
```
